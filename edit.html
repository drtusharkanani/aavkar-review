// ═══════════════════════════════════════════════ // STATE // ═══════════════════════════════════════════════ let currentRecordId = null; let currentDoctorId = null; let editTags = []; // ═══════════════════════════════════════════════ // SCREEN MANAGER // ═══════════════════════════════════════════════ function showScreen(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(id).classList.add('active'); window.scrollTo({ top: 0, behavior: 'smooth' }); } // ═══════════════════════════════════════════════ // SEND OTP — now via /api/otp?action=send // No Airtable token, no Make.com webhook // ═══════════════════════════════════════════════ async function sendOTP() { const id = document.getElementById('loginId').value.trim(); const email = document.getElementById('loginEmail').value.trim().toLowerCase(); if (!id) { showToast('⚠️ Please enter your Doctor ID'); return; } if (!email) { showToast('⚠️ Please enter your email'); return; } showScreen('loadingScreen'); try { const res = await fetch('/api/otp?action=send', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, doctorId: id }), }); const data = await res.json(); if (!res.ok) { showToast('❌ ' + (data.error || 'Doctor ID / email not found')); showScreen('loginScreen'); return; } document.getElementById('otpSubtext').textContent = `OTP sent to ${email}`; showScreen('otpScreen'); showToast('📧 OTP sent! Check your email'); } catch(e) { showToast('⚠️ Network error. Try again.'); showScreen('loginScreen'); } } // ═══════════════════════════════════════════════ // VERIFY OTP — now via /api/otp?action=verify // Returns doctorId + recordId on success // ═══════════════════════════════════════════════ async function verifyOTP() { const otp = document.getElementById('otpInput').value.trim(); const email = document.getElementById('loginEmail').value.trim().toLowerCase(); if (!otp || otp.length !== 6) { showToast('⚠️ Please enter 6-digit OTP'); return; } showScreen('loadingScreen'); try { const res = await fetch('/api/otp?action=verify', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, otp }), }); const data = await res.json(); if (!res.ok || !data.valid) { showToast('❌ ' + (data.error || 'Wrong OTP. Try again.')); showScreen('otpScreen'); return; } // OTP verified — load doctor profile from API currentRecordId = data.recordId; currentDoctorId = data.doctorId; const docRes = await fetch(`/api/doctor?id=${data.doctorId}`); const docData = await docRes.json(); if (!docRes.ok) { showToast('⚠️ Could not load profile. Try again.'); showScreen('loginScreen'); return; } loadProfile(docData); showScreen('profileScreen'); showToast('✅ Verified! Profile loaded.'); } catch(e) { showToast('⚠️ Network error. Try again.'); showScreen('otpScreen'); } } // ═══════════════════════════════════════════════ // LOAD PROFILE INTO FORM // ═══════════════════════════════════════════════ function loadProfile(r) { document.getElementById('e_name').value = r.name || ''; document.getElementById('e_degree').value = r.degree || ''; document.getElementById('e_specialty').value = r.specialty|| ''; document.getElementById('e_hospital').value = r.hospital || ''; document.getElementById('e_city').value = r.city || ''; document.getElementById('e_phone').value = r.phone || ''; document.getElementById('e_gmb').value = r.gmbUrl || ''; // Status card document.getElementById('st_id').textContent = r.id || currentDoctorId; document.getElementById('st_plan').textContent = (r.plan || 'free').replace(/_/g,' ').toUpperCase(); document.getElementById('st_expiry').textContent = r.expiryDate || 'As per plan'; document.getElementById('st_link').href = `https://goodreview.in/review.html?id=${r.id || currentDoctorId}`; const isActive = r.active !== false; document.getElementById('st_status').innerHTML = isActive ? 'ACTIVE ✅' : 'INACTIVE ❌'; // Custom tags — only for premium/ultimate const plan = r.plan || 'free'; const hasTags = plan.includes('premium') || plan.includes('ultimate'); document.getElementById('customTagCard').style.display = hasTags ? 'block' : 'none'; editTags = r.customTags || []; renderEditTags(); } // ═══════════════════════════════════════════════ // CUSTOM TAGS EDITOR // ═══════════════════════════════════════════════ function addEditTag() { if (editTags.length >= 10) { showToast('⚠️ Max 10 tags allowed'); return; } const emoji = document.getElementById('e_tagEmoji').value.trim() || '🏷️'; const label = document.getElementById('e_tagLabel').value.trim(); if (!label) { showToast('⚠️ Please enter tag name'); return; } editTags.push({ emoji, label }); renderEditTags(); document.getElementById('e_tagEmoji').value = ''; document.getElementById('e_tagLabel').value = ''; } function removeEditTag(idx) { editTags.splice(idx, 1); renderEditTags(); } function renderEditTags() { const container = document.getElementById('e_tagsPreview'); container.innerHTML = editTags.map((t, i) => `
${t.emoji}${t.label}×
` ).join(''); document.getElementById('e_tagCount').textContent = editTags.length + ' / 10 tags'; } // ═══════════════════════════════════════════════ // SAVE CHANGES — now via /api/update-doctor // No Airtable token exposed // ═══════════════════════════════════════════════ async function saveChanges() { if (!currentRecordId) { showToast('⚠️ Not logged in'); return; } const gmb = document.getElementById('e_gmb').value.trim(); if (!gmb) { showToast('⚠️ Google review link cannot be empty'); return; } const fields = { Degree: document.getElementById('e_degree').value.trim(), Specialty: document.getElementById('e_specialty').value, Hospital: document.getElementById('e_hospital').value.trim(), City: document.getElementById('e_city').value.trim(), Phone: document.getElementById('e_phone').value.trim(), 'GMB URL': gmb, CustomTags: JSON.stringify(editTags), }; try { const res = await fetch('/api/update-doctor', { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ recordId: currentRecordId, fields }), }); if (!res.ok) throw new Error('Save failed'); document.getElementById('viewPageBtn').href = `https://goodreview.in/review.html?id=${currentDoctorId}`; showScreen('successScreen'); } catch(e) { showToast('⚠️ Save failed. Please try again.'); } } // ═══════════════════════════════════════════════ // LOGOUT // ═══════════════════════════════════════════════ function logout() { currentRecordId = null; currentDoctorId = null; editTags = []; document.getElementById('loginId').value = ''; document.getElementById('loginEmail').value = ''; document.getElementById('otpInput').value = ''; showScreen('loginScreen'); } // ═══════════════════════════════════════════════ // TOAST // ═══════════════════════════════════════════════ function showToast(msg) { const t = document.getElementById('toast'); t.textContent = msg; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 3000); } // Allow Enter on tag label document.getElementById('e_tagLabel').addEventListener('keypress', e => { if (e.key === 'Enter') { e.preventDefault(); addEditTag(); } }); window.sendOTP = sendOTP; window.verifyOTP = verifyOTP; window.saveChanges = saveChanges; window.logout = logout; window.addEditTag = addEditTag; window.removeEditTag = removeEditTag; window.showScreen = showScreen;
